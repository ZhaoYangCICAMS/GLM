\begin{align*}
&\mbox{{\it Any} $n\times p$ matrix $X$ can be written}\\
&X = QR,\\
&\mbox{where $Q$ is an $n\times p$ matrix with orthonormal columns}\\
&\mbox{and $R$ is a $p \times p$ triangular matrix.}\\
\\
&Q Q^T y \mbox{ is an orthogonal projection of y}.
\end{align*}
