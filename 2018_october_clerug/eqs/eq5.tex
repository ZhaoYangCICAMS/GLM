\begin{align*}
&\mbox{{\it Any} n-element vector y can be written}\\
&y = \gamma_1 q_1 + \gamma_2 q_2 + \cdots + \gamma_n q_n,\\
&\mbox{in terms of orthonormal basis vectors $q_i$, and}
&\|y\|^2 = \gamma_1^2 + \gamma_2^2 + \cdots + \gamma_n^2.
\end{align*}
